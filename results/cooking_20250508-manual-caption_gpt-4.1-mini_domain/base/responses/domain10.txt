(define (domain cooking)
    (:requirements :strips)
    (:types robot object location vegetable tool)
    (:predicates
        (holding ?robot - robot ?obj - object)
        (free ?robot - robot)
        (on ?obj - object ?loc - location)
        (sliced ?veg - vegetable)
        (has_tool ?robot - robot ?tool - tool)
        (knife ?tool - tool)
        (bowl ?obj - object)
        (cutting_board ?loc - location)
        (counter ?loc - location)
        (vegetable ?obj - vegetable)
    )
    (:action pick
        :parameters (?robot - robot ?obj - object ?pick_loc - location)
        :precondition (and (on ?obj ?pick_loc) (free ?robot))
        :effect (and (holding ?robot ?obj) (not (free ?robot)) (not (on ?obj ?pick_loc)))
    )
    (:action place
        :parameters (?robot - robot ?obj - object ?place_loc - location)
        :precondition (holding ?robot ?obj)
        :effect (and (on ?obj ?place_loc) (free ?robot) (not (holding ?robot ?obj)))
    )
    (:action slice
        :parameters (?robot - robot ?veg - vegetable ?tool - tool ?loc - location)
        :precondition (and (holding ?robot ?tool) (knife ?tool) (on ?veg ?loc) (cutting_board ?loc))
        :effect (sliced ?veg)
    )
)
