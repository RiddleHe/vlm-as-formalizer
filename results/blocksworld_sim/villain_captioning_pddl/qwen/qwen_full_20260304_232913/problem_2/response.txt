```pddl
(define (problem blocksworld_goal)
    (:domain blocksworld)
    (:objects
        pink - block
        purple - block
        blue - block
        red - block
        robot - robot
    )
    (:init
        (ontable blue)
        (on purple blue)
        (on pink purple)
        (clear pink)
        (ontable red)
        (clear red)
        (handempty robot)
    )
    (:goal (and
        (ontable pink)
        (on purple pink)
        (on blue purple)
        (on red blue)
        (clear red)
    ))
)
```
