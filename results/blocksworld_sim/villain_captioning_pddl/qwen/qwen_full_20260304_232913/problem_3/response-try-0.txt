(define (problem blocksworld_task)
    (:domain blocksworld)
    (:objects
        yellow - block
        green - block
        pink - block
        red - block
        purple - block
        robot1 - robot
    )
    (:init
        (ontable yellow)
        (ontable green)
        (ontable pink)
        (ontable red)
        (on purple red)
        (clear yellow)
        (clear green)
        (clear pink)
        (handempty robot1)
    )
    (:goal (and
        (on yellow green)
        (on green pink)
        (on red purple)
        (ontable pink)
        (ontable purple)
    ))
)
