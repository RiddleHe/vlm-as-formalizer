(define (problem block_stack_goal)
    (:domain blocksworld)
    (:objects
        green_block - block
        yellow_block - block
        pink_block - block
        red_block - block
        robot1 - robot
    )
    (:init
        (ontable green_block)
        (ontable yellow_block)
        (ontable pink_block)
        (ontable red_block)
        (clear green_block)
        (clear yellow_block)
        (clear pink_block)
        (clear red_block)
        (handempty robot1)
    )
    (:goal (and
        (on pink_block red_block)
        (on red_block yellow_block)
        (on yellow_block green_block)
        (ontable green_block)
        (clear pink_block)
    ))
)
