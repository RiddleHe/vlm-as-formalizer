(define (problem blocksworld-problem)
    (:domain blocksworld)
    (:objects
        blue green red yellow orange purple robot1 - robot
    )
    (:init
        (handempty robot1)
        (ontable blue)
        (ontable green)
        (ontable red)
        (ontable yellow)
        (ontable orange)
        (ontable purple)
        (clear blue)
        (clear green)
        (clear red)
        (clear yellow)
        (clear orange)
        (clear purple)
    )
    (:goal (and
        (on blue green)
        (on green red)
        (on red yellow)
        (on yellow orange)
        (on orange purple)
    ))
)
