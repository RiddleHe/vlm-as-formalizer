(define (problem blocksworld3)
    (:domain blocksworld)
    (:objects
        red orange yellow green purple blue robot1 - robot
    )
    (:init
        (ontable red)
        (ontable orange)
        (ontable yellow)
        (ontable green)
        (ontable purple)
        (ontable blue)
        (clear red)
        (clear orange)
        (clear yellow)
        (clear green)
        (clear purple)
        (clear blue)
        (handempty robot1)
    )
    (:goal (and
        (on yellow orange)
        (on purple green)
        (on red blue)
    ))
)
