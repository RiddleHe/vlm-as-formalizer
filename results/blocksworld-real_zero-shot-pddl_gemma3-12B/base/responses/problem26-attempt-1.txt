(define (problem blocksworld3)
    (:domain blocksworld)
    (:objects
        green blue purple orange red yellow - block
        robot1 - robot
    )
    (:init
        (ontable yellow)
        (ontable orange)
        (ontable blue)
        (ontable green)
        (ontable purple)
        (clear yellow)
        (clear orange)
        (clear blue)
        (clear green)
        (clear purple)
        (handempty robot1)
    )
    (:goal (and
        (on green blue)
        (on purple orange)
        (on red yellow)
    ))
)
