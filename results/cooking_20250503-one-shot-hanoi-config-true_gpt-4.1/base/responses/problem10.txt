```lisp
(define (problem cooking-slice-and-place)
    (:domain cooking)
    (:objects
        robot1 robot2 - robot
        tomato1 - vegetable
        carrot1 - vegetable
        knife1 - tool
        cutting_board bowl1 counter - location
    )
    (:init
        (free robot1)
        (carry robot2 knife1)
        (can-cut knife1)

        (is-whole tomato1)
        (not (is-sliced tomato1))
        (at tomato1 counter)
        (available tomato1)

        (not (is-whole carrot1))
        (is-sliced carrot1)
        (at carrot1 cutting_board)
        (available carrot1)

        (at knife1 robot2)
        (is-workspace cutting_board)
        (is-workspace counter)
        (at bowl1 counter)
    )
    (:goal (and
        (is-sliced tomato1)
        (at tomato1 bowl1)
        (at carrot1 bowl1)
    ))
)
```
