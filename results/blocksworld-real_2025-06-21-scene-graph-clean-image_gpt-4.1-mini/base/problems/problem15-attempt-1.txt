    (:objects 
        red green yellow orange blue purple - block
        robot1 - robot
    )
    (:init
        (ontable red)
        (on green red)
        (on orange green)
        (on purple orange)
        (on blue purple)

        (clear blue)
        (handempty robot1)
    )
    (:goal (and

        (on blue purple)
        (on purple orange)

    ))
