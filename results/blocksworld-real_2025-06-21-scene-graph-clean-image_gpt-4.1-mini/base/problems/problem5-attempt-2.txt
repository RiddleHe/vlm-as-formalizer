(define (problem stack_blocks)
    (:domain blocksworld)
    (:objects
        red yellow orange purple blue green - block
        robot1 - robot
    )
    (:init
        (ontable purple)
        (ontable red)
        (ontable orange)
        (ontable green)
        (clear yellow)
        (clear blue)
        (clear purple)
        (clear green)
        (handempty robot1)
        (on yellow purple)
        (on blue red)
        (on orange green)
    )
    (:goal (and
        (on yellow purple)
        (on blue red)
        (on orange green)
    ))
)
