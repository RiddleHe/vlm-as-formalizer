(define (problem blocksworld_task)
    (:domain blocksworld)
    (:objects
        orange red yellow green blue purple - block
        robot1 - robot
    )
    (:init
        (ontable orange)
        (ontable yellow)
        (ontable blue)
        (ontable purple)
        (clear red)
        (clear green)
        (clear orange)
        (clear purple)
        (handempty robot1)
        (on green blue)
        (on purple orange)
        (on red yellow)
    )
    (:goal (and
        (on green blue)
        (on purple orange)
        (on red yellow)
    ))
)
