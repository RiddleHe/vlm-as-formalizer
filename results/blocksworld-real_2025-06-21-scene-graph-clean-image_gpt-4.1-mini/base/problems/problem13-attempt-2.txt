(define (problem blocksworld_stacks)
  (:domain blocksworld)
  (:objects
    blue_block - block
    yellow_block - block
    orange_block - block
    red_block - block
    purple_block - block
    green_block - block
    robot1 - robot
  )
  (:init
    (ontable yellow_block)
    (ontable green_block)
    (on red_block purple_block)
    (on purple_block green_block)
    (clear blue_block)
    (clear orange_block)
    (handempty robot1)
  )
  (:goal (and
    (on blue_block yellow_block)
    (on orange_block red_block)
    (on red_block purple_block)
    (on purple_block green_block)
  ))
)
