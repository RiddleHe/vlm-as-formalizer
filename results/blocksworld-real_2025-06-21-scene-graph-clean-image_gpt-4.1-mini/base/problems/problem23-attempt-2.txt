(define (problem blocksworld_real_1)
    (:domain blocksworld)
    (:objects
        blue green yellow purple orange red - block
        robot1 - robot
    )
    (:init
        (ontable blue)
        (ontable purple)
        (clear yellow)
        (clear red)
        (handempty robot1)
        (on yellow blue)
        (on orange green)
        (on green purple)
        (on red orange)
    )
    (:goal (and
        (on yellow blue)
        (on red orange)
        (on orange green)
        (on green purple)
    ))
)
