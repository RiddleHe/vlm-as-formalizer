(define (problem blocksworld_stack_blocks)
    (:domain blocksworld)
    (:objects
        blue green purple red orange yellow
        robot1 - robot
    )
    (:init
        (ontable blue)
        (ontable green)
        (ontable purple)
        (ontable red)
        (clear blue)
        (clear green)
        (clear yellow)
        (clear orange)
        (clear red)
        (handempty robot1)
    )
    (:goal (and
        (on yellow orange)
        (on orange red)
        (on red purple)
        (on blue green)
    ))
)
