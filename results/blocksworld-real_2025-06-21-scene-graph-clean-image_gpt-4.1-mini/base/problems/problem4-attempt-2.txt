(define (problem stack_blocks)
    (:domain blocksworld)
    (:objects
        green_block purple_block blue_block red_block orange_block yellow_block - block
        robot_arm - robot
    )
    (:init
        (ontable green_block)
        (ontable blue_block)
        (ontable red_block)
        (ontable orange_block)
        (ontable yellow_block)

        (clear purple_block)
        (clear yellow_block)
        (clear green_block)

        (handempty robot_arm)
    )
    (:goal (and
        (on purple_block green_block)
        (on blue_block red_block)
        (on red_block orange_block)
        (on orange_block yellow_block)
    ))
)
