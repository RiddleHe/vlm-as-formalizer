(define (problem blocksworld3)
    (:domain blocksworld)
    (:objects
        yellow red blue green purple orange - block
        robot1 - robot
    )
    (:init
        (ontable yellow)
        (ontable red)
        (ontable blue)
        (ontable green)
        (ontable purple)
        (ontable orange)
        (clear yellow)
        (clear red)
        (clear blue)
        (clear purple)
        (clear orange)
        (clear green)
        (handempty robot1)
    )
    (:goal (and
        (on yellow orange)
        (on red green)
        (on blue purple)
    ))
)
