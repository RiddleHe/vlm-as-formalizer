(define (problem blocksworld_problem1)
    (:domain blocksworld)
    (:objects
        red blue green yellow orange purple - block
        robot1 - robot
    )
    (:init
        (ontable purple)
        (ontable orange)
        (on red blue)
        (on green purple)
        (on yellow red)
        (on orange yellow)
        (clear blue)
        (clear red)
        (clear orange)
        (handempty robot1)
    )
    (:goal (and
        (on orange yellow)
        (on yellow red)
        (on red blue)
        (on blue green)
        (on green purple)
    ))
)
