(define (problem blocksworld_stack2)
    (:domain blocksworld)
    (:objects
        blue orange purple yellow red green
        robot1 - robot
    )
    (:init
        (ontable yellow)
        (ontable red)
        (ontable green)
        (ontable orange)
        (on purple blue)
        (on red green)
        (on orange purple)
        (clear blue)
        (clear orange)
        (clear yellow)
        (handempty robot1)
    )
    (:goal (and 
        (on orange purple)
        (on purple blue)
        (on yellow red)
        (on red green)
    ))
)
