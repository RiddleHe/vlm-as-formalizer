(define (problem stack_blocks)
    (:domain blocksworld)
    (:objects
        orange red green purple yellow blue - block
        robot1 - robot
    )
    (:init
        (ontable orange)
        (ontable red)
        (ontable green)
        (ontable purple)
        (ontable yellow)
        (ontable blue)
        (clear orange)
        (clear red)
        (clear green)
        (clear purple)
        (clear yellow)
        (clear blue)
        (handempty robot1)
    )
    (:goal (and
        (on purple yellow)
        (on yellow orange)
        (on blue red)
        (on red green)
    ))
)
