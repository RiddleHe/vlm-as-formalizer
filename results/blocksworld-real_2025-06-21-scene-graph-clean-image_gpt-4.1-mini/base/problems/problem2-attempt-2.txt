(define (problem blocksworld_problem)
    (:domain blocksworld)
    (:objects
        orange blue purple red green yellow - block
        robot1 - robot
    )
    (:init
        (ontable yellow)
        (clear green)
        (on orange blue)
        (on blue purple)
        (on purple red)
        (ontable red)
        (ontable green)
        (handempty robot1)
    )
    (:goal (and 
        (on green red)
        (on red purple)
        (on purple blue)
        (on blue orange)
        (on orange yellow)
    ))
)
