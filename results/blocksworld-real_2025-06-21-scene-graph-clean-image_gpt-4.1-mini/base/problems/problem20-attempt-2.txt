(define (problem blocksworld_stack_blocks)
    (:domain blocksworld)
    (:objects
        blue green purple red orange yellow - block
        robot1 - robot
    )
    (:init
        (ontable blue)
        (ontable green)
        (ontable purple)
        (ontable red)
        (ontable orange)
        (ontable yellow)
        (clear blue)
        (clear green)
        (handempty robot1)
    )
    (:goal (and
        (on yellow orange)
        (on orange red)
        (on red purple)
        (on blue green)
    ))
)
