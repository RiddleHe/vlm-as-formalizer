(define (problem stack_green_over_red_over_orange_over_purple_over_blue_over_yellow)
  (:domain blocksworld)
  (:objects
    purple orange blue yellow red green - block
    robot1 - robot
  )
  (:init
    (ontable blue)
    (ontable yellow)
    (ontable red)
    (ontable green)
    (clear purple)
    (clear yellow)
    (clear red)
    (clear green)
    (handempty robot1)
    (on purple orange)
    (on orange blue)
  )
  (:goal (and
    (on green red)
    (on red orange)
    (on orange purple)
    (on purple blue)
    (on blue yellow)
  ))
)
