```pddl
(define (problem blocksworld-task)
    (:domain blocksworld)
    (:objects
        orange - block
        yellow - block
        green - block
        red - block
        purple - block
        blue - block
        robot1 - robot
    )
    (:init
        (clear orange)
        (ontable orange)
        (ontable yellow)
        (ontable green)
        (ontable red)
        (ontable purple)
        (clear green)
        (clear red)
        (handempty robot1)
    )
    (:goal
        (and 
            (on blue orange)
            (on yellow green)
            (on red yellow)
            (on purple red)
        )
    )
)
```
