(define (problem blocksworld-problem)
    (:domain blocksworld)
    (:objects
        purple - block
        yellow - block
        blue - block
        red - block
        green - block
        orange - block
        robot1 - robot
    )
    (:init
        (handempty robot1)
        (ontable purple)
        (clear purple)
        (ontable yellow)
        (clear yellow)
        (on blue yellow)
        (clear blue)
        (ontable red)
        (clear red)
        (ontable green)
        (clear green)
        (ontable orange)
        (clear orange)
    )
    (:goal (and
        (on blue green)
        (on green red)
        (on red yellow)
        (on yellow orange)
        (on orange purple)
    ))
)
