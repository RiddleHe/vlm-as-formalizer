(define (problem blocksworld-problem)
    (:domain blocksworld)
    (:objects
        blue purple yellow green red orange robot1 - block robot
    )
    (:init
        (ontable blue)
        (on blue purple)
        (on purple yellow)
        (ontable green)
        (on green red)
        (on red orange)
        (clear yellow)
        (clear orange)
        (handempty robot1)
    )
    (:goal (and
        (on blue yellow)
        (on orange red)
        (on red purple)
        (on purple green)
    ))
