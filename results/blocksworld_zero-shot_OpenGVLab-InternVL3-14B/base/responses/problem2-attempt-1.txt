        pink - block
        purple - block
        blue - block

    (:init

        (ontable blue)

        (ontable red)
        (handempty robot)
        (clear pink)
        (clear blue)
        (not (handfull robot))
        (not (holding pink))
        (not (holding red))
        (not (holding purple))
        (not (holding blue))
