(define (problem stack_blocks)
    (:domain blocksworld)
    (:objects
        yellow red orange green purple blue - block
        robot1 - robot
    )
    (:init
        (ontable yellow)
        (ontable green)
        (clear orange)
        (clear blue)
        (clear purple)
        (clear yellow)
        (on red yellow)
        (on orange red)
        (on purple green)
        (on blue purple)
        (handempty robot1)
    )
    (:goal (and
        (on orange red)
        (on red yellow)
        (ontable yellow)
        (on blue purple)
        (on purple green)
        (ontable green)
    ))
)
