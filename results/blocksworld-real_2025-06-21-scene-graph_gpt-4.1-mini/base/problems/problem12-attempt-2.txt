(define (problem stack_blocks)
    (:domain blocksworld)
    (:objects
        blue red orange yellow green purple - block
        robot1 - robot
    )
    (:init
        (ontable blue)
        (ontable orange)
        (ontable green)
        (clear yellow)
        (clear purple)
        (clear red)
        (clear blue)
        (clear orange)
        (clear green)
        (handempty robot1)
    )
    (:goal (and
        (on yellow orange)
        (on purple green)
        (on red blue)
    ))
)
