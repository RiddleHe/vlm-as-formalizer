(define (problem stack_blocks)
    (:domain blocksworld)
    (:objects
        orange blue yellow green red purple
        robot1 - robot
    )
    (:init
        (ontable purple)
        (ontable red)
        (ontable orange)
        (ontable yellow)
        (on red purple)
        (on green red)
        (on blue orange)
        (clear blue)
        (clear yellow)
        (clear orange)
        (handempty robot1)
    )
    (:goal (and
        (on blue orange)
        (on green red)
        (on red purple)
        (on yellow green)
    ))
)
