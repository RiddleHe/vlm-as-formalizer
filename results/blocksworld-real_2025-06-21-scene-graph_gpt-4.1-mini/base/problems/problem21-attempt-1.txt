(define (problem stack_three)
    (:domain blocksworld)
    (:objects
        green red orange purple blue yellow - block
        robot1 - robot
    )
    (:init
        (ontable red)
        (ontable purple)
        (ontable yellow)
        (clear green)
        (clear orange)
        (clear blue)
        (clear red)
        (clear purple)
        (clear yellow)
        (on green red)
        (on orange purple)
        (on blue yellow)
        (handempty robot1)
    )
    (:goal (and 
        (on green red)
        (on orange purple)
        (on blue yellow)
    ))
)
