(define (problem blocksworld_real_1)
    (:domain blocksworld)
    (:objects
        purple blue green yellow orange red
        robot1 - robot
    )
    (:init
        (ontable purple)
        (ontable blue)
        (ontable yellow)
        (ontable orange)
        (clear green)
        (clear purple)
        (clear red)
        (handempty robot1)
        (on green blue)
        (on red yellow)
    )
    (:goal (and
        (on green blue)
        (on purple orange)
        (on red yellow)
    ))
)
