(define (problem blocksworld_problem)
    (:domain blocksworld)
    (:objects
        purple red yellow orange green blue
        robot1 - robot
    )
    (:init
        (ontable red)
        (ontable green)
        (ontable blue)
        (clear purple)
        (clear yellow)
        (handempty robot1)
        (on purple red)
        (on yellow orange)
        (on orange green)
        (on green blue)
    )
    (:goal (and
        (on purple red)
        (on yellow orange)
        (on orange green)
        (on green blue)
    ))
)
