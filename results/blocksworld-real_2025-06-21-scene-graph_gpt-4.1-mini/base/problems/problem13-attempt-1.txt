(define (problem blocksworld_task)
    (:domain blocksworld)
    (:objects
        blue yellow orange red purple green
        robot1 - robot
    )
    (:init
        (ontable yellow)
        (ontable green)
        (clear blue)
        (clear orange)
        (clear green)
        (clear yellow)
        (handempty robot1)
        (on blue yellow)
        (on orange red)
        (on red purple)
        (on purple green)
    )
    (:goal (and 
        (on blue yellow)
        (on orange red)
        (on red purple)
        (on purple green)
    ))
)
