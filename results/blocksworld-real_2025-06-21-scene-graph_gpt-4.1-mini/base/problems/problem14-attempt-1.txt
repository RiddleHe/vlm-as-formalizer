(define (problem stack_blocks)
    (:domain blocksworld)
    (:objects
        purple yellow orange blue red green - block
        robot1 - robot
    )
    (:init
        (ontable orange)
        (ontable green)
        (clear purple)
        (clear blue)
        (clear red)
        (clear yellow)
        (handempty robot1)
        (on purple yellow)
        (on yellow orange)
        (on blue red)
        (on red green)
    )
    (:goal (and
        (on purple yellow)
        (on yellow orange)
        (on blue red)
        (on red green)
    ))
)
