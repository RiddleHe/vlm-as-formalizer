(define (problem stack_all_blocks)
    (:domain blocksworld)
    (:objects 
        orange yellow purple blue green red - block
        robot1 - robot
    )
    (:init
        (ontable green)
        (ontable yellow)
        (ontable red)
        (clear blue)
        (clear orange)
        (handempty robot1)
        (on orange purple)
        (on blue orange)
        (on red green)
        (on purple red)
        (on green yellow)
    )
    (:goal (and
        (on blue orange)
        (on orange red)
        (on red purple)
        (on purple green)
        (on green yellow)
        (ontable yellow)
    ))
)
