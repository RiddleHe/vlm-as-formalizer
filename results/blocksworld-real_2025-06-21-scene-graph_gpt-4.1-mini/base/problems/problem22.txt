(define (problem blocksworld_problem)
    (:domain blocksworld)
    (:objects
        orange_block red_block blue_block green_block purple_block yellow_block - block
        robot1 - robot
    )
    (:init
        (ontable orange_block)
        (ontable red_block)
        (ontable blue_block)
        (ontable green_block)
        (ontable purple_block)
        (ontable yellow_block)
        (clear orange_block)
        (clear red_block)
        (clear blue_block)
        (clear green_block)
        (clear purple_block)
        (clear yellow_block)
        (handempty robot1)
    )
    (:goal (and 
        (on green_block orange_block)
        (on purple_block yellow_block)
        (on yellow_block red_block)
        (on red_block blue_block)
    ))
)
