(define (problem blockstacking)
    (:domain blocksworld)
    (:objects
        green yellow red orange purple blue - block
        robot1 - robot
    )
    (:init
        (ontable purple)
        (ontable red)
        (ontable blue)
        (clear orange)
        (clear blue)
        (clear red)
        (clear purple)
        (clear green)
        (handempty robot1)
        (not (handfull robot1))
        (not (holding orange))
        (not (holding yellow))
        (not (holding green))
        (not (holding purple))
        (not (holding red))
        (not (holding blue))
        (not (on orange yellow))
        (not (on yellow green))
        (not (on green purple))
        (not (on purple red))
        (not (on red blue))
    )
    (:goal (and
        (on orange yellow)
        (on yellow green)
        (on green purple)
        (on purple red)
        (on red blue)
    ))
)
