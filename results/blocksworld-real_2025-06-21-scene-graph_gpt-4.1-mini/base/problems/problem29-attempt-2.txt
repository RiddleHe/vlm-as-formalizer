(define (problem blockstacking)
    (:domain blocksworld)
    (:objects
        blue orange green purple red yellow - block
        robot1 - robot
    )
    (:init
        (ontable yellow)
        (clear blue)
        (handempty robot1)
        (on blue orange)
        (on orange green)
        (on green purple)
        (on purple red)
        (on red yellow)
    )
    (:goal (and
        (on blue orange)
        (on orange green)
        (on green purple)
        (on purple red)
        (on red yellow)
    ))
)
