(define (problem stack_blocks)
    (:domain blocksworld)
    (:objects
        blue purple orange yellow red green
        robot1 - robot
    )
    (:init
        (ontable blue)
        (ontable red)
        (ontable green)
        (clear orange)
        (clear yellow)
        (clear blue)
        (on orange purple)
        (on purple blue)
        (on yellow red)
        (on red green)
        (handempty robot1)
    )
    (:goal (and 
        (on orange purple)
        (on purple blue)
        (ontable blue)
        (on yellow red)
        (on red green)
        (ontable green)
    ))
)
