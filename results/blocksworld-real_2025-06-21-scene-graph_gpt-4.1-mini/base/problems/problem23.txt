(define (problem blocksworld_problem)
    (:domain blocksworld)
    (:objects
        blue green yellow purple red orange - block
        robot1 - robot
    )
    (:init
        (ontable yellow)
        (ontable orange)
        (clear blue)
        (clear purple)
        (handempty robot1)
        (on blue green)
        (on green yellow)
        (on purple red)
        (on red orange)
    )
    (:goal (and
        (on yellow blue)
        (on orange green)
        (on green purple)
        (on red orange)
    ))
)
