(define (problem blocksworld_problem1)
    (:domain blocksworld)
    (:objects
        blue purple orange green red yellow - block
        robot1 - robot
    )
    (:init
        (ontable purple)
        (ontable orange)
        (ontable yellow)
        (on green blue)
        (on blue purple)
        (on red orange)
        (clear green)
        (clear red)
        (clear yellow)
        (handempty robot1)
    )
    (:goal (and 
        (on green blue) 
        (on blue purple) 
        (on purple orange) 
        (on red yellow)
    ))
)
