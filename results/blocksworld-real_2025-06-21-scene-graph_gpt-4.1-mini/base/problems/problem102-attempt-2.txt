(define (problem stack_blocks)
    (:domain blocksworld)
    (:objects
        green red orange blue purple yellow - block
        robot1 - robot
    )
    (:init
        (ontable yellow)
        (clear green)
        (handempty robot1)
        (on green red)
        (on red orange)
        (on orange blue)
        (on blue purple)
        (on purple yellow)
    )
    (:goal (and
        (on green red)
        (on red orange)
        (on orange blue)
        (on blue purple)
        (on purple yellow)
        (ontable yellow)
    ))
)
