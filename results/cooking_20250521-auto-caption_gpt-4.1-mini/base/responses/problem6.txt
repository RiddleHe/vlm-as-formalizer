PDDL problem:
(define (problem slice-vegetables)
    (:domain cooking)
    (:objects
        robot1
        robot2
        knife - tool
        cutting_board bowl - location
        cucumber carrot - vegetable
    )
    (:init
        (free robot1)
        (carry robot2 knife)
        (can-cut knife)
        (is-whole cucumber)
        (is-whole carrot)
        (not (is-sliced cucumber))
        (not (is-sliced carrot))
        (available cucumber)
        (available carrot)
        (at cucumber cutting_board)
        (at carrot cutting_board)
        (at knife robot2)
        (at cutting_board cutting_board)
        (at bowl bowl)
        (is-workspace cutting_board)
    )
    (:goal (and
        (is-sliced cucumber)
        (is-sliced carrot)
        (at cucumber bowl)
        (at carrot bowl)
    ))
)
