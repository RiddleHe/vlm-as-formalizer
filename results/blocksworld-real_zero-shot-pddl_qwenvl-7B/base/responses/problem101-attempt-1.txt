(define (problem blocksworld)

(:requirements :strips :typing)

    yellow_block1 - block
    yellow_block2 - block
    orange_block1 - block
    orange_block2 - block
    red_block1 - block
    red_block2 - block
    green_block1 - block
    green_block2 - block
    blue_block1 - block
    blue_block2 - block
    purple_block1 - block
    purple_block2 - block
    robot1 - robot
)
(:init
    (clear yellow_block1)
    (clear yellow_block2)
    (clear orange_block1)
    (clear orange_block2)
    (clear red_block1)
    (clear red_block2)
    (clear green_block1)
    (clear green_block2)
    (clear blue_block1)
    (clear blue_block2)
    (clear purple_block1)
    (clear purple_block2)

    (ontable yellow_block2)

    (ontable green_block1)
    (ontable green_block2)
    (ontable blue_block1)
    (ontable blue_block2)
    (ontable purple_block1)
    (ontable purple_block2)

)
(:goal (and
    (on yellow_block1 orange_block1)
    (on orange_block1 orange_block2)
    (on red_block1 green_block1)
    (on green_block1 green_block2)
    (on blue_block1 purple_block1)

))
)
