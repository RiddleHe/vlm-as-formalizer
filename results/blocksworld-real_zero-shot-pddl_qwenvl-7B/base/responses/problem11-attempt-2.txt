```pddl
(define (problem blocksworld)
(:domain blocksworld)
(:objects
    green - block
    blue - block
    purple - block
    orange - block
    red - block
    yellow - block
    robot1 - robot
)
(:init
    (ontable green)
    (ontable blue)
    (ontable purple)
    (ontable orange)
    (ontable red)
    (ontable yellow)
    (handempty robot1)
)
(:goal
    (and
        (on green blue)
        (on blue purple)
        (on purple orange)
        (on red yellow)
    )
)
)
```
