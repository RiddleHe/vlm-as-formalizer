    (:objects
        block_blue - block
        block_green - block
        block_orange - block
        block_purple - block

        block_yellow - block

        (ontable block_green)

        (ontable block_yellow)
        (clear block_blue)
        (clear block_green)
        (clear block_orange)
        (clear block_purple)
        (clear block_red)
        (clear block_yellow)
        (handempty robot_arm)

    (:goal (and

        (on block_red block_yellow)

        (clear block_red)
        (ontable block_orange)
        (ontable block_yellow)
    ))
