(define (problem blocksworld_task)
    (:domain blocksworld)
    (:objects
        green - block
        blue - block
        yellow - block
        purple - block
        orange - block
        red - block
        robot - robot
    )
    (:init
        (on green blue)
        (on blue yellow)
        (on yellow purple)
        (ontable purple)
        (ontable orange)
        (ontable red)
        (clear green)
        (clear orange)
        (clear red)
        (handempty robot)
    )
    (:goal (and
        (on yellow red)
        (on red orange)
        (on orange blue)
        (on blue purple)
        (on purple green)
        (ontable green)
        (clear yellow)
    ))
)
