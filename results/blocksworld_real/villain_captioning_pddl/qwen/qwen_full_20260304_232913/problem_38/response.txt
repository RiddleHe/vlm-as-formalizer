```pddl
(define (problem blocksworld-goal)
    (:domain blocksworld)
    (:objects
        red - block
        yellow - block
        purple - block
        blue - block
        green - block
        orange - block
        robot1 - robot
    )
    (:init
        (on red yellow)
        (on yellow purple)
        (ontable purple)
        (clear red)
        (on green orange)
        (ontable orange)
        (clear green)
        (ontable blue)
        (clear blue)
        (handempty robot1)
    )
    (:goal (and
        (on green blue)
        (on blue red)
        (on red yellow)
        (on yellow purple)
        (on purple orange)
        (ontable orange)
        (clear green)
    ))
)
```
