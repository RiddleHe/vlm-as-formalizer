```pddl
(define (problem blocksworld_goal)
    (:domain blocksworld)
    (:objects
        purple - block
        yellow - block
        orange - block
        blue - block
        red - block
        green - block
        robot1 - robot
    )
    (:init
        (on orange red)
        (on red green)
        (ontable green)
        (clear orange)
        (on purple yellow)
        (on yellow blue)
        (ontable blue)
        (clear purple)
        (handempty robot1)
    )
    (:goal (and
        (on purple yellow)
        (on yellow orange)
        (ontable orange)
        (clear purple)
        (on blue red)
        (on red green)
        (ontable green)
        (clear blue)
    ))
)
```
