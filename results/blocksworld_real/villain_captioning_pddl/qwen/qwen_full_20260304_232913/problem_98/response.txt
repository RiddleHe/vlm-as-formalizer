(define (problem blocksworld_rearrange)
    (:domain blocksworld)
    (:objects
        yellow_block - block
        purple_block - block
        green_block - block
        orange_block1 - block
        blue_block - block
        orange_block2 - block
        robot1 - robot
    )
    (:init
        (on yellow_block purple_block)
        (on purple_block green_block)
        (on green_block orange_block2)
        (on orange_block2 blue_block)
        (on blue_block orange_block1)
        (ontable orange_block1)
        (clear yellow_block)
        (handempty robot1)
    )
    (:goal (and
        (on orange_block1 orange_block2)
        (on yellow_block blue_block)
        (on purple_block green_block)
        (ontable orange_block2)
        (ontable blue_block)
        (ontable green_block)
        (clear orange_block1)
        (clear yellow_block)
        (clear purple_block)
    ))
)
