```pddl
(define (problem blocksworld-goal)
    (:domain blocksworld)
    (:objects
        yellow red green blue orange purple - block
        robot1 - robot
    )
    (:init
        (on yellow red)
        (on red green)
        (on blue orange)
        (ontable orange)
        (ontable green)
        (ontable purple)
        (clear yellow)
        (clear blue)
        (clear purple)
        (handempty robot1)
    )
    (:goal (and
        (on orange red)
        (on red yellow)
        (on blue purple)
        (on purple green)
        (ontable yellow)
        (ontable green)
        (clear orange)
        (clear blue)
    ))
)
```
