```lisp
(define (problem blocks_stacks)
    (:domain blocksworld)
    (:objects
        robot1 - robot

        orange yellow green blue red purple - block
    )
    (:init
        ;; Initial stacks (each one block on table and clear)
        (ontable orange)
        (clear orange)

        (ontable yellow)
        (clear yellow)

        (ontable green)
        (clear green)

        (ontable blue)
        (clear blue)

        (ontable red)
        (clear red)

        (ontable purple)
        (clear purple)

        ;; robot is empty-handed
        (handempty robot1)
    )
    (:goal (and
        (on blue green)
        (on red yellow)
        (on purple orange)
    ))
)
```
