(define (problem stack_blocks)
    (:domain blocksworld)
    (:objects
        orange yellow blue purple red green - block
        robot1 - robot
    )
    (:init
        ; stack 1: orange on table, clear
        (ontable orange)
        (clear orange)
        ; stack 2: yellow on table, clear
        (ontable yellow)
        (clear yellow)
        ; stack 3: blue on purple, blue clear
        (on blue purple)
        (clear blue)
        (ontable purple)
        (not (clear purple)) ; purple has blue on top
        ; stack 4: red on table, clear
        (ontable red)
        (clear red)
        ; stack 5: green on table, clear
        (ontable green)
        (clear green)

        ; robot is empty-handed
        (handempty robot1)
    )
    (:goal (and
        (on blue orange)
        (on orange red)
        (on red purple)
        (on purple green)
        (on green yellow)
    ))
)
