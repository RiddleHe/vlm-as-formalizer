```lisp
(define (problem kitchen1)
(:domain cooking)
(:objects
    robot1 - robot
    cucumber - vegetable
    carrot - vegetable
    tomato - vegetable
    knife - tool
    bowl - location
    cutting_board - location
    counter - location
)
(:init
    (available cucumber)
    (available carrot)
    (available tomato)
    (is-whole cucumber)
    (is-whole carrot)
    (is-whole tomato)
    (free robot1)
    (at cucumber counter)
    (at carrot counter)
    (at tomato counter)
    (at knife cutting_board)
    (is-workspace cutting_board)
)
(:goal (and 
    (at cucumber bowl) 
    (is-sliced cucumber) 
    (at carrot bowl) 
    (is-sliced carrot) 
    (at tomato bowl) 
    (is-sliced tomato)
))
)
```
