(define (problem stack_two_piles)
    (:domain blocksworld)
    (:objects
        blue yellow orange red purple green - block
        robot1 - robot
    )
    (:init
        (ontable blue)
        (ontable yellow)
        (ontable orange)
        (ontable red)
        (ontable purple)
        (ontable green)
        (clear blue)
        (clear yellow)
        (clear orange)
        (clear red)
        (clear purple)
        (clear green)
        (handempty robot1)
    )
    (:goal (and
        (on blue yellow)
        (ontable yellow)
        (on orange red)
        (on red purple)
        (on purple green)
        (ontable green)
    ))
)
