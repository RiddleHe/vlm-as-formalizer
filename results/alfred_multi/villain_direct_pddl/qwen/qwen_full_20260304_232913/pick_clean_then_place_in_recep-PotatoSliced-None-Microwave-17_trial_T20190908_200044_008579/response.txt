```pddl
(define (problem put_task_problem)
    (:domain put_task)
    (:objects
        agent1 - agent
        microwave1 - microwave
        sink1 - sink
        knife1 - knife
        potato1 - object
        table1 - receptacle
    )
    (:init
        (atLocation agent1 table1)
        (inReceptacle potato1 table1)
        (inReceptacle knife1 table1)
        (openable microwave1)
        (openable sink1)
        (not (opened microwave1))
        (not (opened sink1))
        (not (isClean potato1))
        (not (isSliced potato1))
        (not (isHot potato1))
        (not (holdsAny agent1))
    )
    (:goal
        (and
            (inReceptacle potato1 microwave1)
            (isClean potato1)
            (isSliced potato1)
        )
    )
)
```
