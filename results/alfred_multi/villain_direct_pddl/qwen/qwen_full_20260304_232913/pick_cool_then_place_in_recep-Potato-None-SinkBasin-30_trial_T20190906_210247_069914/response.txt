(define (problem put_cold_brown_ball_in_sink)
    (:domain put_task)
    (:objects
        agent1 - agent
        fridge1 - fridge
        sink1 - sink
        brown_ball1 - object
        knife1 - knife
    )
    (:init
        (atLocation agent1 fridge1)
        (inReceptacle brown_ball1 fridge1)
        (openable fridge1)
        (not (opened fridge1))
        (not (holdsAny agent1))
        (not (isCool brown_ball1))
        (not (isClean brown_ball1))
        (not (isSliced brown_ball1))
        (not (isHot brown_ball1))
        (not (isOn brown_ball1))
        (not (isToggled brown_ball1))
    )
    (:goal
        (and
            (inReceptacle brown_ball1 sink1)
            (isCool brown_ball1)
        )
    )
)
