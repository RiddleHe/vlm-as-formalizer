```pddl
(define (problem put_box_plate_on_table)
    (:domain put_task)
    (:objects
        agent1 - agent
        box - object
        plate - object
        table - receptacle
    )
    (:init
        (atLocation agent1 box)
        (inReceptacle plate box)
        (not (atLocation agent1 table))
    )
    (:goal (and
        (inReceptacle plate table)
        (inReceptacle box table)
    ))
)
```
