```pddl
(define (problem put_box_on_chair)
    (:domain put_task)
    (:objects
        robot - agent
        cardboard_box - object
        green_chair - receptacle
    )
    (:init
        (atLocation robot cardboard_box)
        (openable green_chair)
        (not (opened green_chair))
        (not (holdsAny robot))
    )
    (:goal
        (inReceptacle cardboard_box green_chair)
    )
)
```
