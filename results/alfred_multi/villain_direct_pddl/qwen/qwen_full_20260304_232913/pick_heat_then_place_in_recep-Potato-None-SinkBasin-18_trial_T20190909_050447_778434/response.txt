(define (problem put_heated_potato_in_sink)
    (:domain put_task)
    (:objects
        agent1 - agent
        microwave1 - microwave
        sink1 - sink
        potato1 - object
        knife1 - knife
    )
    (:init
        (atLocation agent1 microwave1)
        (inReceptacle potato1 microwave1)
        (openable microwave1)
        (opened microwave1)
        (isOn microwave1)
        (isToggled microwave1)
        (not (isHot potato1))
        (not (isClean potato1))
        (not (isSliced potato1))
        (not (holdsAny agent1))
        (not (holds agent1 potato1))
    )
    (:goal (and
        (inReceptacle potato1 sink1)
        (isHot potato1)
    ))
)
