```pddl
(define (problem put_task_problem)
    (:domain put_task)
    (:objects
        agent1 - agent
        statue1 statue2 - object
        coffee_table - receptacle
    )
    (:init
        (atLocation agent1 statue1)
        (atLocation agent1 statue2)
        (atLocation agent1 coffee_table)
        (not (holdsAny agent1))
    )
    (:goal
        (and
            (inReceptacle statue1 coffee_table)
            (inReceptacle statue2 coffee_table)
        )
    )
)
```
