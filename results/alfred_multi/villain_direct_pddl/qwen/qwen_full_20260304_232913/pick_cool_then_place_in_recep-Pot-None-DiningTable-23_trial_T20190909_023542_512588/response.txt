```pddl
(define (problem put_chilled_bowl_on_table)
    (:domain put_task)
    (:objects
        agent1 - agent
        fridge1 - fridge
        table1 - receptacle
        bowl1 - object
    )
    (:init
        (openable fridge1)
        (not (opened fridge1))
        (inReceptacle bowl1 fridge1)
        (isCool bowl1)
        (not (atLocation agent1 fridge1))
        (not (atLocation agent1 table1))
        (not (holdsAny agent1))
    )
    (:goal
        (and
            (inReceptacle bowl1 table1)
            (atLocation agent1 table1)
        )
    )
)
```
