(define (problem put_task_problem)
    (:domain put_task)
    (:objects
        agent1 - agent
        fridge1 - fridge
        microwave1 - microwave
        sink1 - sink
        potato1 - object
        knife1 - knife
        pan1 - object
        countertop1 - receptacle
        stove1 - receptacle
    )
    (:init
        (openable fridge1)
        (openable microwave1)
        (atLocation agent1 countertop1)
        (inReceptacle potato1 countertop1)
        (inReceptacle knife1 countertop1)
        (inReceptacle pan1 stove1)
        (not (isSliced potato1))
        (not (isHot potato1))
        (not (isCool potato1))
        (not (opened fridge1))
        (not (opened microwave1))
        (not (holdsAny agent1))
    )
    (:goal
        (and
            (inReceptacle potato1 fridge1)
            (isSliced potato1)
            (isHot potato1)
            (isCool potato1)
        )
    )
)
