```pddl
(define (problem put_box_with_plate_on_table)
    (:domain put_task)
    (:objects
        agent1 - agent
        table - receptacle
        box - object
        plate - object
        microwave - receptacle
        statuette - object
        newspaper - object
        keys - object
    )
    (:init
        (not (atLocation agent1 table))
        (not (atLocation agent1 box))
        (not (atLocation agent1 microwave))
        (not (atLocation agent1 plate))
        (not (holdsAny agent1))
        (not (opened microwave))
        (openable microwave)
        (inReceptacle plate table)
        (inReceptacle statuette table)
        (inReceptacle newspaper table)
        (inReceptacle keys table)
        (not (isClean plate))
        (not (isHot plate))
        (not (isCool plate))
        (not (isSliced plate))
        (not (isOn microwave))
        (not (isToggled microwave))
    )
    (:goal
        (and
            (inReceptacle plate box)
            (inReceptacle box table)
        )
    )
)
```
