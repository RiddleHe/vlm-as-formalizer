(define (problem chill_mug_and_place_on_rack)
    (:domain put_task)
    (:objects
        agent1 - agent
        mug - object
        fridge - fridge
        rack - receptacle
        coffee_maker - object
        knife - knife
    )
    (:init
        (not (atLocation agent1 mug))
        (not (atLocation agent1 fridge))
        (not (atLocation agent1 rack))
        (not (atLocation agent1 coffee_maker))
        (not (atLocation agent1 knife))
        (not (holdsAny agent1))
        (not (holds agent1 mug))
        (not (isCool mug))
        (not (isHot mug))
        (not (isClean mug))
        (not (isSliced mug))
        (not (isOn mug))
        (not (isToggled mug))
        (not (opened fridge))
        (openable fridge)
        (inReceptacle mug coffee_maker)
        (atLocation mug coffee_maker)
        (atLocation knife coffee_maker)
        (atLocation fridge coffee_maker)
        (atLocation rack coffee_maker)
    )
    (:goal (and
        (inReceptacle mug rack)
        (isCool mug)
    ))
)
