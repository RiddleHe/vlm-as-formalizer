```pddl
(define (problem cool_cup)
    (:domain put_task)
    (:objects
        agent1 - agent
        fridge1 - fridge
        coffee_maker1 - object
        cup1 - object
    )
    (:init
        (openable fridge1)
        (not (opened fridge1))
        (not (inReceptacle cup1 fridge1))
        (not (holdsAny agent1))
        (not (atLocation agent1 coffee_maker1))
        (not (atLocation agent1 fridge1))
        (isOn coffee_maker1)
        (isToggled coffee_maker1)
    )
    (:goal (and
        (isCool cup1)
    ))
)
```
