(define (problem cut_bread_fridge_task)
    (:domain put_task)
    (:objects
        agent1 - agent
        fridge1 - fridge
        bread1 - object
        apple1 - object
        knife1 - knife
    )
    (:init
        (openable fridge1)
        (not (opened fridge1))
        (not (atLocation agent1 fridge1))
        (not (atLocation agent1 bread1))
        (not (atLocation agent1 apple1))
        (not (atLocation agent1 knife1))
        (not (holdsAny agent1))
        (not (isSliced bread1))
        (not (inReceptacle bread1 fridge1))
        (not (isCool bread1))
    )
    (:goal (and
        (inReceptacle bread1 fridge1)
        (isSliced bread1)
        (not (inReceptacle bread1 fridge1))
        (atLocation agent1 bread1)
        (atLocation agent1 apple1)
    ))
)
