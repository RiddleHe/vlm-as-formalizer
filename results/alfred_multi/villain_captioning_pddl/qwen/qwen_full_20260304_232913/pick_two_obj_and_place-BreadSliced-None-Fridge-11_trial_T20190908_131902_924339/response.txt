    (:domain put_task)
    (:objects
        agent1 - agent
        bread - object
        knife - knife
        fridge - fridge
        table - receptacle
        countertop - receptacle
        microwave - microwave
        sink - sink
        stove - object
    )
    (:init

        (inReceptacle bread table)

        (openable fridge)
        (not (opened fridge))
        (not (isSliced bread))
        (not (holdsAny agent1))
        (not (inReceptacle bread fridge))
    )

        (isSliced bread)
        (opened fridge)
    ))
