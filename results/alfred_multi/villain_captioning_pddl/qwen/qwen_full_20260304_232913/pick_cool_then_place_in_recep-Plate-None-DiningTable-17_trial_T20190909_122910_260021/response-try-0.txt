(define (problem put_chilled_plate_on_table)
    (:domain put_task)
    (:objects
        robot - agent
        fridge - fridge
        table - object
        plate - object
        bread - object
        tomato - object
        lettuce - object
        knife - knife
        fork - object
        spoon - object
        salt_shaker - object
        pepper_shaker - object
    )
    (:init
        (openable fridge)
        (not (opened fridge))
        (not (holdsAny robot))
        (not (isCool plate))
        (inReceptacle plate fridge)
        (atLocation robot table) ; assuming robot starts near table for initial navigation
    )
    (:goal
        (and
            (inReceptacle plate table)
            (isCool plate)
        )
    )
)
