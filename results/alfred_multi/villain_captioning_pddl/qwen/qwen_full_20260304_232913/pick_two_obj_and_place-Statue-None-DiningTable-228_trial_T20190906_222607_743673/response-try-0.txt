(define (problem set_statues_on_table)
    (:domain put_task)
    (:objects
        agent1 - agent
        statue1 statue2 remote_control - object
        coffee_table sofa - receptacle
    )
    (:init
        (atLocation agent1 coffee_table)
        (inReceptacle statue1 coffee_table)
        (inReceptacle statue2 coffee_table)
        (inReceptacle remote_control coffee_table)
        (not (holdsAny agent1))
    )
    (:goal (and
        (inReceptacle statue1 coffee_table)
        (inReceptacle statue2 coffee_table)
    ))
)
