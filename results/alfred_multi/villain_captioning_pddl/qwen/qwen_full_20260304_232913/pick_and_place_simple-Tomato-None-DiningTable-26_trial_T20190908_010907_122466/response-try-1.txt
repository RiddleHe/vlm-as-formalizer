(define (problem find_tomato_on_table)
    (:domain put_task)
    (:objects
        agent1 - agent
        tomato1 - object
        bread1 - object
        spatula1 - object
        cup1 - object
        box1 - object
        table1 - object
        fridge1 - fridge
    )
    (:init
        (atLocation agent1 table1)
        (inReceptacle tomato1 table1)
        (inReceptacle bread1 table1)
        (inReceptacle spatula1 table1)
        (inReceptacle cup1 table1)
        (inReceptacle box1 table1)
        (openable fridge1)
        (not (opened fridge1))
        (not (holdsAny agent1))
        (not (isSliced tomato1))
        (not (isClean tomato1))
        (not (isHot tomato1))
        (not (isCool tomato1))
        (not (isToggled tomato1))
    )
    (:goal (and (inReceptacle tomato1 table1)))
)
