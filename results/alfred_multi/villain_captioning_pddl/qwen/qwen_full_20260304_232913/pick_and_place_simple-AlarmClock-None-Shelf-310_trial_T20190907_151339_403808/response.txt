(define (problem put_clock_on_desk)
    (:domain put_task)
    (:objects
        agent1 - agent
        desk - receptacle
        cabinet - receptacle
        clock - object
        smartphone - object
        lamp - object
    )
    (:init
        (openable cabinet)
        (not (opened cabinet))
        (inReceptacle clock desk)
        (inReceptacle smartphone desk)
        (isOn lamp)
        (not (holdsAny agent1))
        (not (atLocation agent1 clock))
        (not (atLocation agent1 smartphone))
        (not (atLocation agent1 lamp))
        (not (atLocation agent1 desk))
        (not (atLocation agent1 cabinet))
        (not (isClean clock))
        (not (isHot clock))
        (not (isCool clock))
        (not (isSliced clock))
    )
    (:goal (and
        (inReceptacle clock desk)
        (atLocation agent1 clock)
    ))
)
