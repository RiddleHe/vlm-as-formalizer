(define (problem chill_lettuce)
    (:domain put_task)
    (:objects
        agent1 - agent
        fridge1 - fridge
        microwave1 - microwave
        sink1 - sink
        lettuce1 lettuce2 - object
        knife1 - knife
        spoon1 - object
        wineglass1 - object
        mug1 - object
        saltshaker1 saltshaker2 - object
        book1 - object
        container1 - object
        table1 - object
        chair1 - object
    )
    (:init
        (openable fridge1)
        (openable microwave1)
        (not (opened fridge1))
        (not (opened microwave1))
        (not (isCool lettuce1))
        (not (isCool lettuce2))
        (not (isSliced lettuce1))
        (not (isSliced lettuce2))
        (not (holdsAny agent1))
        (inReceptacle lettuce1 table1)
        (inReceptacle lettuce2 table1)
        (inReceptacle knife1 table1)
        (inReceptacle spoon1 table1)
        (inReceptacle wineglass1 table1)
        (inReceptacle mug1 table1)
        (inReceptacle saltshaker1 table1)
        (inReceptacle saltshaker2 table1)
        (inReceptacle book1 table1)
        (inReceptacle container1 table1)
        (inReceptacle chair1 table1)
        (not (atLocation agent1 fridge1))
        (not (atLocation agent1 microwave1))
        (not (atLocation agent1 sink1))
        (not (atLocation agent1 lettuce1))
        (not (atLocation agent1 lettuce2))
        (not (atLocation agent1 knife1))
        (not (atLocation agent1 spoon1))
        (not (atLocation agent1 wineglass1))
        (not (atLocation agent1 mug1))
        (not (atLocation agent1 saltshaker1))
        (not (atLocation agent1 saltshaker2))
        (not (atLocation agent1 book1))
        (not (atLocation agent1 container1))
        (not (atLocation agent1 table1))
        (not (atLocation agent1 chair1))
    )
    (:goal (and
        (inReceptacle lettuce1 table1)
        (isCool lettuce1)
    ))
)
