(define (problem put_cold_brown_ball_in_sink)
    (:domain put_task)
    (:objects
        agent1 - agent
        fridge1 - fridge
        sink1 - sink
        brown_ball1 - object
        knife1 - knife
        fork1 - object
        soap_dispenser1 - object
        plant_pot1 - object
        bowl1 - object
        recycling_bin1 - object
    )
    (:init
        (openable fridge1)
        (not (opened fridge1))
        (inReceptacle brown_ball1 recycling_bin1)
        (atLocation agent1 recycling_bin1) ; Assuming agent starts near the bin for accessibility
        (not (isCool brown_ball1))
        (not (holdsAny agent1))
        (not (holds agent1 brown_ball1))
        (not (inReceptacle brown_ball1 sink1))
        (not (isClean brown_ball1))
        (not (isHot brown_ball1))
        (not (isSliced brown_ball1))
        (not (isOn brown_ball1))
        (not (isToggled brown_ball1))
        (atLocation agent1 recycling_bin1)
    )
    (:goal (and
        (inReceptacle brown_ball1 sink1)
        (isCool brown_ball1)
    ))
)
