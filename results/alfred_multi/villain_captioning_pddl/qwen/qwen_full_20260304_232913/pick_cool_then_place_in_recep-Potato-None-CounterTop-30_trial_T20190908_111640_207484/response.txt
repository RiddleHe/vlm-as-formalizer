(define (problem put_chilled_potato_on_island)
    (:domain put_task)
    (:objects
        agent1 - agent
        fridge1 - fridge
        microwave1 - microwave
        countertop1 - receptacle
        potato1 - object
        knife1 - knife
        bread1 - object
        green_veg1 - object
        shaker1 - object
        bowl1 - object
        tray1 - object
    )
    (:init
        (openable fridge1)
        (not (opened fridge1))
        (openable microwave1)
        (not (opened microwave1))
        (not (isCool potato1))
        (not (isSliced potato1))
        (not (isHot potato1))
        (not (holdsAny agent1))
        (not (atLocation agent1 countertop1))
        (not (atLocation agent1 fridge1))
        (not (atLocation agent1 microwave1))
        (inReceptacle potato1 countertop1)
        (inReceptacle knife1 countertop1)
        (inReceptacle bread1 countertop1)
        (inReceptacle green_veg1 countertop1)
        (inReceptacle shaker1 countertop1)
        (inReceptacle bowl1 countertop1)
        (inReceptacle tray1 countertop1)
        (not (isOn microwave1))
    )
    (:goal (and
        (inReceptacle potato1 countertop1)
        (isCool potato1)
    ))
)
