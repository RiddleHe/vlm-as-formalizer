(define (problem rinse_pan)
    (:domain put_task)
    (:objects
        agent1 - agent
        pan - object
        sink - receptacle
        microwave - receptacle
        egg - object
    )
    (:init
        (atLocation agent1 pan)
        (not (holdsAny agent1))
        (not (isClean pan))
        (openable sink)
        (not (opened sink))
        (not (inReceptacle pan sink))
        (not (holds agent1 pan))
        (atLocation agent1 pan)
        (atLocation agent1 microwave)
        (atLocation agent1 sink)
        (atLocation agent1 egg)
    )
    (:goal (and (isClean pan)))
)
