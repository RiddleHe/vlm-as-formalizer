```pddl
(define (problem put_clean_egg_on_counter)
    (:domain put_task)
    (:objects
        agent1 - agent
        microwave1 - microwave
        sink1 - sink
        counter1 - receptacle
        egg1 - object
    )
    (:init
        (openable microwave1)
        (not (opened microwave1))
        (inReceptacle egg1 microwave1)
        (not (isClean egg1))
        (not (isHot egg1))
        (not (isCool egg1))
        (not (isSliced egg1))
        (not (isOn microwave1))
        (not (isToggled microwave1))
        (not (atLocation agent1 microwave1))
        (not (atLocation agent1 sink1))
        (not (atLocation agent1 counter1))
        (not (holdsAny agent1))
    )
    (:goal (and
        (inReceptacle egg1 counter1)
        (isClean egg1)
    ))
)
```
