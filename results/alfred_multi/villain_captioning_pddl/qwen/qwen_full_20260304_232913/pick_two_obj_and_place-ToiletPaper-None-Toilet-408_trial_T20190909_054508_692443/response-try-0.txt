(define (problem put_toilet_paper_on_toilet)
    (:domain put_task)
    (:objects
        agent1 - agent
        toilet_paper_full - object
        toilet_paper_empty - object
        trash_can - receptacle
        toilet - object
    )
    (:init
        (openable trash_can)
        (not (opened trash_can)) ; Assuming it's accessible but not formally "opened" like a lid
        (inReceptacle toilet_paper_full trash_can)
        (inReceptacle toilet_paper_empty trash_can)
        (not (holdsAny agent1))
        (not (atLocation agent1 trash_can))
        (not (atLocation agent1 toilet))
    )
    (:goal (and
        (inReceptacle toilet_paper_full toilet) ; Treating toilet as a surface/receptacle for placement
        (inReceptacle toilet_paper_empty toilet) ; Treating toilet as a surface/receptacle for placement
    ))
)
