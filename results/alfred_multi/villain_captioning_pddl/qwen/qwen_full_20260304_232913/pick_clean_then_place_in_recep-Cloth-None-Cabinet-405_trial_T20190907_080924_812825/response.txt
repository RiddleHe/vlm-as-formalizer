(define (problem put_rinsed_cloth_in_cabinet)
    (:domain put_task)
    (:objects
        agent1 - agent
        sink1 - sink
        cabinet1 - receptacle
        cloth1 - object
        candle1 - object
        spray_bottle1 - object
        soap1 - object
        trash_bin1 - object
    )
    (:init
        (not (atLocation agent1 cloth1))
        (not (atLocation agent1 sink1))
        (not (atLocation agent1 cabinet1))
        (not (holdsAny agent1))
        (not (isClean cloth1))
        (not (opened cabinet1))
        (openable cabinet1)
        (inReceptacle soap1 sink1)
    )
    (:goal (and
        (inReceptacle cloth1 cabinet1)
        (isClean cloth1)
        (opened cabinet1)
    ))
)
