```lisp
(define (problem blocksworld-stacks-3)
  (:domain blocksworld)
  (:objects
    block-green block-red block-orange block-purple block-blue block-yellow - block
    robot1 - robot
  )
  (:init
    ;; robot state
    (handempty robot1)

    ;; Stack A: green on red, red on table
    (on block-green block-red)
    (ontable block-red)
    (clear block-green)

    ;; Stack B: blue on orange on purple on yellow, yellow on table
    (on block-blue block-orange)
    (on block-orange block-purple)
    (on block-purple block-yellow)
    (ontable block-yellow)
    (clear block-blue)
  )
  (:goal
    (and
      (on block-green block-red)
      (on block-orange block-purple)
      (on block-blue block-yellow)
    )
  )
)
```
