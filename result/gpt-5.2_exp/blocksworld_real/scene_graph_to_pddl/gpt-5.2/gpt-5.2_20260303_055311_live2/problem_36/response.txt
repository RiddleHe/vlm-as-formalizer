  (:objects
    green-block purple-block yellow-block orange-block red-block blue-block - block
    robot-arm - robot
  )
  (:init
    ;; on-table blocks
    (ontable green-block)
    (ontable purple-block)

    (clear orange-block)
    (clear blue-block)

  (:goal
